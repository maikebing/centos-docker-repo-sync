快速配置

1下载 repo 配置文件

curl -o /etc/yum.repos.d/uixe_mirror.repo http://mirrors.uixe.net/uixe_mirror.repo

2备份并禁用原有源

# 备份原有 repo 文件
mkdir -p /etc/yum.repos.d/backup
mv /etc/yum.repos.d/CentOS-*.repo /etc/yum.repos.d/backup/ 2>/dev/null

3刷新缓存

yum clean all
yum makecache

4验证仓库

yum repolist

域名 hosts文件设置

     echo "10.165.79.130  git.uixe.net registry.uixe.net  mirrors.uixe.net"  >> /etc/hosts

一键配置（复制粘贴即可）

    curl -o /etc/yum.repos.d/uixe_mirror.repo http://mirrors.uixe.net/uixe_mirror.repo \
&& mkdir -p /etc/yum.repos.d/backup \
&& mv /etc/yum.repos.d/CentOS-*.repo /etc/yum.repos.d/backup/ 2>/dev/null \
&& yum clean all \
&& yum makecache

安装 Docker

# 安装 Docker CE
yum install -y docker-ce docker-ce-cli containerd.io

#docker-compose 插件
rm /usr/local/bin/docker-compose  -f;rm /usr/local/sbin/docker-compose -f;rm /usr/bin/docker-compose -f
ln -s /usr/libexec/docker/cli-plugins/docker-compose /usr/local/bin/docker-compose

# 启动 Docker
systemctl start docker
systemctl enable docker

# 验证安装
docker --version
